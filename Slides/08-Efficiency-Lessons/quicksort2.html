Quicksort: inductive proof of correctness 

      ;;; quicksort : ListOfX (X X -> Boolean) -> ListOfX
      ;;;
      ;;; Given a list of things
      ;;;     and a total order on those things,
      ;;; returns a sorted list of the things.
      
      (define (quicksort things less?)
        (if (< (length things) 2)
            things
            (let ((pivot (first things))
                  (other (rest things)))
              (let-values (((below above)
                            (partition (lambda (x)
                                         (less? x pivot))
                                       other)))
                (append (quicksort below less?)
                        (list pivot)
                        (quicksort above less?))))))
